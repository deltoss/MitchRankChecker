112  public async Task<ActionResult<RankCheckRequest>> PostRankCheckRequest(RankCheckRequest rankCheckRequest)
113  {
114  rankCheckRequest.Status = RankCheckRequestStatus.InQueue;
115  _context.RankCheckRequests.Add(rankCheckRequest);
116  await _context.SaveChangesAsync();
117 
118  _queue.QueueBackgroundWorkItem(async (token, provider) =>
119  {
120  RankCheckRequest updatedRankCheckRequest;
121  var context = provider.GetService<RankCheckerDbContext>();
122 
123  updatedRankCheckRequest = await context.RankCheckRequests.Where(x => x.Id == rankCheckRequest.Id).FirstAsync();
124  updatedRankCheckRequest.Status = RankCheckRequestStatus.InProgress;
125  context.Attach(updatedRankCheckRequest);
126  context.Entry(updatedRankCheckRequest).Property(x => x.StatusId).IsModified = true; // Mark the specific field as modified to only update that field
127  await context.SaveChangesAsync();
128  try
129  {
130  var scrapingRankCheckerFactory = provider.GetService<IRankCheckerFactory>();
131  IRankChecker rankChecker = scrapingRankCheckerFactory.CreateRankChecker(updatedRankCheckRequest);
132  List<SearchEntry> relevantSearchEntries = await rankChecker.ExtractRankEntriesAsync();
133  if (relevantSearchEntries != null && relevantSearchEntries.Count > 0)
134  await context.SearchEntries.AddRangeAsync(relevantSearchEntries);
135 
136  updatedRankCheckRequest.Status = RankCheckRequestStatus.Completed;
137  context.Attach(updatedRankCheckRequest);
138  context.Entry(updatedRankCheckRequest).Property(x => x.StatusId).IsModified = true;
139  await context.SaveChangesAsync();
140  }
141  catch (Exception ex)
142  {
143  updatedRankCheckRequest.Status = RankCheckRequestStatus.Error;
144  updatedRankCheckRequest.ErrorMessage = ex.Message;
145  context.Attach(updatedRankCheckRequest);
146  context.Entry(updatedRankCheckRequest).Property(x => x.StatusId).IsModified = true;
147  context.Entry(updatedRankCheckRequest).Property(x => x.ErrorMessage).IsModified = true;
148  await context.SaveChangesAsync();
149  }
150  });
151 
152  return CreatedAtAction("GetRankCheckRequest", new { id = rankCheckRequest.Id }, rankCheckRequest);
153  }
